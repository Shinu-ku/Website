      #include <iostream>
      using namespace std;

      int main() {
        char choice;
        float value1, value2;

        cout << "----- Basic Calculator--------" << endl;
        cout << "Enter your choice : \n + : Addition \n - : Subtraction \n / : Division \n * : Multiplication" << endl;
        cout << "your input__";
        cin >> choice;
        cout << "Enter the value1: ";
        cin >> value1;
        cout << "Enter the value2: ";
        cin >> value2;

        switch(choice) {
          case '+':
            cout << "Sum of value1 and value2 is " << (value1 + value2) << endl;
            break;
          case '-':
            cout << "Difference of value1 and value2 is " << (value1 - value2) << endl;
            break;
          case '/:
            cout << "Division of value1 and value2 is " << (value1 / value2) << endl;
            break;
          case '*':
            cout << "Multiplication of value1 and value2 is " << (value1 * value2) << endl;
            break;
          default:
            cout << "Invalid Input!" << endl;
        }

        return 0;
      }
