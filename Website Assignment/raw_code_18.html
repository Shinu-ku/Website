      #include <iostream>
      using namespace std;

      int main()
      {
          int n, sum = 0;
          cout << "Enter a number: ";
          cin >> n;

          for (int i = 1; i <= n; i++)
          {
              sum += i;
          }

          cout << "Sum of numbers from 1 to " << n << " is: " << sum << endl;

          return 0;
      }
Code Explanation:
  • This program calculates the sum of all numbers from 1 to the number entered by the user.
  • The user is prompted to enter a number, and then the program calculates the sum from 1 to that number using a for loop.
  • It then displays the sum.

(Note: The output depends on the user's input. For example, if the input is 5, the output will be "Sum of numbers from 1 to 5 is: 15".)
